002package org.openstreetmap.josm.plugins.pointinfo;
003
004import static org.openstreetmap.josm.tools.I18n.tr;
005
006import java.awt.GridBagConstraints;
007import java.awt.GridBagLayout;
008
009import javax.swing.Box;
010import javax.swing.JCheckBox;
011import javax.swing.JComboBox;
012import javax.swing.JLabel;
013import javax.swing.JPanel;
014
015import org.openstreetmap.josm.Main;
016import org.openstreetmap.josm.gui.preferences.DefaultTabPreferenceSetting;
017import org.openstreetmap.josm.gui.preferences.PreferenceTabbedPane;
018import org.openstreetmap.josm.tools.GBC;
019
020/**
021 * Plugin preferences.
022 */
023public class PointInfoPreference extends DefaultTabPreferenceSetting {
024
025    private final JComboBox<String> module = new JComboBox<>();
026    private final JCheckBox autoMode = new JCheckBox(tr("Automatically detect the module"));
027
028    /**
029     * Constructs a new {@code PointInfoPreference}.
030     */
031    public PointInfoPreference() {
032        super("pointinfo", tr("Point information settings"), tr("Settings for the point information plugin."), true);
033    }
034
035    @Override
036    public String getIconName() {
037        return "info-sml.png";
038    }
039
040    @Override
041    public void addGui(PreferenceTabbedPane gui) {
042        JPanel panel = new JPanel(new GridBagLayout());
043        // autoMode
044        autoMode.setSelected(Main.pref.getBoolean("plugin.pointinfo.automode", false));
045        autoMode.setToolTipText(tr("Try to guess the appropriate module from the location."
046                + " If it fails, use the module selected below."));
047        panel.add(autoMode, GBC.eol().insets(0, 0, 0, 0));
048        // module
049        for (String modName : PointInfoPlugin.getModules()) {
050            module.addItem(modName);
051        }
052        module.setSelectedItem(Main.pref.get("plugin.pointinfo.module", "RUIAN"));
053        module.setToolTipText(tr("The module called to get the point information."));
054        panel.add(new JLabel(tr("Module")), GBC.std());
055        panel.add(module, GBC.eol().fill(GridBagConstraints.HORIZONTAL).insets(5, 0, 0, 5));
056        panel.add(Box.createVerticalGlue(), GBC.eol().fill(GridBagConstraints.VERTICAL));
057        createPreferenceTabWithScrollPane(gui, panel);
058    }
059
060    @Override
061    public boolean ok() {
062        Main.pref.putBoolean("plugin.pointinfo.automode", autoMode.isSelected());
063        Main.pref.put("plugin.pointinfo.module", (String) module.getSelectedItem());
064        return false;
065    }
066}
